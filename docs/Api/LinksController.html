def create
  @link = Link.new(link_params)
  if @link.save
    respond_with(@link, location: api_link_url(@link))
  else
    respond_with(@link.errors, status: :unprocessable_entity)
  end
end

def update
  @link = Link.find(params[:id])
  if @link.update(link_params)
    respond_with(@link)

  end
